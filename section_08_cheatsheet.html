# Parameters  ← Toggle this cell as a parameter cell (3-dot menu → Toggle parameter cell)
p_FullLoad = True
API_URL    = "https://restcountries.com/v3.1/all"
RAW_TABLE  = "raw_countries"

import requests, json
from pyspark.sql import functions as F
from datetime import datetime

# ── Fetch from REST Countries API ─────────────────────────────
print("Fetching REST Countries API...")
r = requests.get(API_URL, timeout=30)
r.raise_for_status()           # raises HTTPError if status >= 400
countries = r.json()
print(f"Fetched {len(countries)} countries")

# ── Save raw JSON to Files/bronze/ (immutable archive) ────────
ts       = datetime.now().strftime("%Y%m%d_%H%M%S")
raw_path = f"Files/bronze/countries/countries_{ts}.json"
mssparkutils.fs.put(raw_path, json.dumps(countries), overwrite=True)
print(f"Saved raw JSON → {raw_path}")

# ── Parse JSON → flat Python dicts ────────────────────────────
def parse(c):
    curs = c.get("currencies", {})
    return {
        "country_code" : c.get("cca2", ""),
        "name_common"  : c.get("name", {}).get("common", ""),
        "region"       : c.get("region", ""),
        "subregion"    : c.get("subregion") or "",      # None-safe
        "population"   : int(c.get("population") or 0), # None-safe
        "area_km2"     : float(c.get("area") or 0),     # None-safe
        "capital"      : c.get("capital", [""])[0] if c.get("capital") else "",
        "currency_code": list(curs.keys())[0] if curs else "",
        "currency_name": list(curs.values())[0].get("name", "") if curs else "",
        "latitude"     : float(c.get("latlng", [0,0])[0]) if c.get("latlng") else 0.0,
        "longitude"    : float(c.get("latlng", [0,0])[1]) if c.get("latlng") else 0.0,
    }

records = [parse(c) for c in countries]

# ── Create Spark DataFrame ─────────────────────────────────────
df = spark.createDataFrame(records)

# ── Add audit columns ──────────────────────────────────────────
df = (df
    .withColumn("_ingestion_ts", F.current_timestamp())
    .withColumn("_source",       F.lit(API_URL))
    .withColumn("_batch_id",     F.lit(ts)))

# ── Data Quality Checks ────────────────────────────────────────
total = df.count()
assert df.filter(F.col("name_common")  == "").count() == 0, "DQ: empty country names"
assert df.filter(F.col("country_code") == "").count() == 0, "DQ: empty country codes"
print(f"[DQ] {total} records — all checks passed")

# ── Write to Delta (Bronze layer) ─────────────────────────────
(df.write
   .format("delta")
   .mode("overwrite")
   .option("overwriteSchema", "true")
   .saveAsTable(RAW_TABLE))
print(f"Written {RAW_TABLE}: {total} rows")
display(spark.table(RAW_TABLE).limit(5))

# Parameters
p_FullLoad = True
SRC_TABLE  = "raw_sales"
DEST_TABLE = "silver_sales"

from pyspark.sql import functions as F
from pyspark.sql.types import DoubleType, IntegerType

# ── Read from Bronze ──────────────────────────────────────────
df = spark.table(SRC_TABLE)

# ── Defensive rename (safe if DFG2 already renamed) ──────────
rmap = {
    "Row ID":"row_id",       "Order ID":"order_id",
    "Order Date":"order_date",  "Ship Date":"ship_date",
    "Ship Mode":"ship_mode",    "Customer ID":"customer_id",
    "Customer Name":"customer_name", "Segment":"segment",
    "Country":"country",        "City":"city",
    "State":"state",            "Postal Code":"postal_code",
    "Region":"region",          "Product ID":"product_id",
    "Category":"category",      "Sub-Category":"sub_category",
    "Product Name":"product_name", "Sales":"sales",
    "Quantity":"quantity",      "Discount":"discount",
    "Profit":"profit"
}
for old, new in rmap.items():
    if old in df.columns:
        df = df.withColumnRenamed(old, new)

# ── Type casts ────────────────────────────────────────────────
df = (df
    .withColumn("sales",    F.col("sales").cast(DoubleType()))
    .withColumn("quantity", F.col("quantity").cast(IntegerType()))
    .withColumn("discount", F.col("discount").cast(DoubleType()))
    .withColumn("profit",   F.col("profit").cast(DoubleType())))

# ── Derived business columns ──────────────────────────────────
df = (df
    .withColumn("order_year",    F.year(F.col("order_date")))
    .withColumn("order_month",   F.month(F.col("order_date")))
    .withColumn("order_quarter", F.quarter(F.col("order_date")))
    .withColumn("days_to_ship",  F.datediff(F.col("ship_date"), F.col("order_date")))
    .withColumn("profit_margin", F.round(
        F.when(F.col("sales") != 0, F.col("profit") / F.col("sales") * 100)
         .otherwise(F.lit(0.0)), 2))
    .withColumn("net_revenue",   F.round(F.col("sales") * (1 - F.col("discount")), 2))
    .dropDuplicates(["order_id", "product_id"])
    .filter(F.col("order_date").isNotNull())
    .withColumn("_ingestion_ts", F.current_timestamp()))

# ── DQ Checks ─────────────────────────────────────────────────
total = df.count()
assert df.filter(F.col("order_id").isNull()).count()   == 0, "DQ: null order_id"
assert df.filter(F.col("order_date").isNull()).count() == 0, "DQ: null order_date"
assert df.filter(F.col("sales").isNull()).count()      == 0, "DQ: null sales"
print(f"[DQ] {total} rows — all checks passed")

# ── Write to Silver ────────────────────────────────────────────
(df.write
   .format("delta")
   .mode("overwrite")
   .option("overwriteSchema", "true")
   .saveAsTable(DEST_TABLE))
print(f"Written {DEST_TABLE}: {total} rows")
display(spark.table(DEST_TABLE)
        .select("order_id","customer_name","sales","profit",
                "profit_margin","days_to_ship","order_year","order_quarter")
        .limit(5))

# Parameters
p_FullLoad = True
SRC_TABLE  = "raw_countries"
DEST_TABLE = "silver_countries"

from pyspark.sql import functions as F

# ── Read from Bronze ──────────────────────────────────────────
df = spark.table(SRC_TABLE)

# ── Clean and standardise ─────────────────────────────────────
df = (df
    .filter(F.col("name_common")  != "")
    .filter(F.col("country_code") != "")
    .withColumn("name_common",   F.trim(F.col("name_common")))
    .withColumn("region",        F.trim(F.col("region")))
    .withColumn("subregion",     F.coalesce(F.trim(F.col("subregion")),  F.lit("Unknown")))
    .withColumn("capital",       F.coalesce(F.col("capital"),            F.lit("N/A")))
    .withColumn("currency_code", F.coalesce(F.col("currency_code"),      F.lit("Unknown")))
    .withColumn("population_m",  F.round(F.col("population") / 1e6, 3))
    .dropDuplicates(["country_code"])
    .withColumn("_ingestion_ts", F.current_timestamp()))

# ── DQ Checks ─────────────────────────────────────────────────
total = df.count()
assert df.filter(F.col("country_code").isNull()).count() == 0, "DQ: null country_code"
assert df.filter(F.col("name_common") == "").count()     == 0, "DQ: empty name_common"
print(f"[DQ] {total} countries — all checks passed")

# ── Write to Silver ────────────────────────────────────────────
(df.write
   .format("delta")
   .mode("overwrite")
   .option("overwriteSchema", "true")
   .saveAsTable(DEST_TABLE))
print(f"Written {DEST_TABLE}: {total} rows")
display(spark.table(DEST_TABLE)
        .select("country_code","name_common","region","subregion",
                "capital","population_m","currency_code")
        .limit(5))

NB-03-GOLD-StarSchema — Complete notebook
dim_date via SQL sequence (1,461 rows) · dim_customer/product via row_number() SK · dim_geography enriched from silver_countries · fact_sales joins all 3 SKs + generates date_key integers
🥇 Gold
⚠️ All chained .withColumn() calls MUST be in outer () ⚠️ Window.orderBy("natural_key") — keeps SKs deterministic ✓ dim_date: exactly 1,461 rows (2014–2017 incl. 2016 leap year) ✓ fact_sales: ~9,994 rows, 15 columns 💡 order_date_key ≠ date_key by name, but same YYYYMMDD integer values
Python NB-03-GOLD-StarSchema · Complete code 
# Parameters
p_FullLoad    = True
SRC_SALES     = "silver_sales"
SRC_COUNTRIES = "silver_countries"

from pyspark.sql import functions as F
from pyspark.sql.types import IntegerType
from pyspark.sql.window import Window

# ── Load Silver tables ─────────────────────────────────────────
s  = spark.table(SRC_SALES)
cn = spark.table(SRC_COUNTRIES)

# ── 1) dim_date — SQL sequence (no source table needed) ───────
dim_date = spark.sql("""
    SELECT
        CAST(date_format(d, 'yyyyMMdd') AS INT) AS date_key,
        d                                        AS date,
        year(d)                                  AS year,
        quarter(d)                               AS quarter,
        month(d)                                 AS month_num,
        date_format(d, 'MMMM')                  AS month_name,
        date_format(d, 'MMM')                    AS month_short,
        (dayofweek(d) IN (1, 7))                 AS is_weekend,
        date_format(d, 'yyyy-MM')               AS year_month,
        concat('Q', quarter(d))                  AS quarter_label
    FROM (
        SELECT explode(
            sequence(date('2014-01-01'), date('2017-12-31'), interval 1 day)
        ) AS d
    )
""")
dim_date.write.format("delta").mode("overwrite").saveAsTable("dim_date")
print(f"dim_date      : {dim_date.count()} rows")   # expected 1461

# ── 2) dim_customer ───────────────────────────────────────────
dim_customer = (s
    .select("customer_id", "customer_name", "segment")
    .dropDuplicates(["customer_id"])
    .withColumn("customer_sk",
        F.row_number().over(Window.orderBy("customer_id")))
    .withColumn("_ts", F.current_timestamp()))
dim_customer.write.format("delta").mode("overwrite").saveAsTable("dim_customer")
print(f"dim_customer  : {dim_customer.count()} rows")

# ── 3) dim_product ────────────────────────────────────────────
dim_product = (s
    .select("product_id", "product_name", "category", "sub_category")
    .dropDuplicates(["product_id"])
    .withColumn("product_sk",
        F.row_number().over(Window.orderBy("product_id")))
    .withColumn("_ts", F.current_timestamp()))
dim_product.write.format("delta").mode("overwrite").saveAsTable("dim_product")
print(f"dim_product   : {dim_product.count()} rows")

# ── 4) dim_geography — enriched from silver_countries ────────
geo = (s
    .select("country", "state", "city", "region", "postal_code")
    .dropDuplicates())

dim_geography = (geo
    .join(
        cn.select("name_common", "country_code", "subregion"),
        geo.country == cn.name_common,
        "left"
    )
    .drop("name_common")
    .withColumn("geo_sk",
        F.row_number().over(Window.orderBy("country", "state", "city")))
    .withColumn("_ts", F.current_timestamp()))
dim_geography.write.format("delta").mode("overwrite").saveAsTable("dim_geography")
print(f"dim_geography : {dim_geography.count()} rows")

# ── 5) fact_sales — join all dimension SKs ───────────────────
dc = spark.table("dim_customer") .select("customer_id",  "customer_sk")
dp = spark.table("dim_product")  .select("product_id",   "product_sk")
dg = spark.table("dim_geography").select("country", "state", "city", "geo_sk")

fact_sales = (s
    .join(dc, "customer_id", "left")
    .join(dp, "product_id",  "left")
    .join(dg, ["country", "state", "city"], "left")
    .withColumn("order_date_key",
        F.date_format(F.col("order_date"), "yyyyMMdd").cast(IntegerType()))
    .withColumn("ship_date_key",
        F.date_format(F.col("ship_date"),  "yyyyMMdd").cast(IntegerType()))
    .withColumn("fact_sk", F.monotonically_increasing_id())
    .select(
        "fact_sk",    "order_id",
        "order_date_key", "ship_date_key",
        "customer_sk", "product_sk", "geo_sk",
        "sales", "quantity", "discount", "profit",
        "days_to_ship", "profit_margin", "net_revenue",
        "_ingestion_ts"))
fact_sales.write.format("delta").mode("overwrite").saveAsTable("fact_sales")
print(f"fact_sales    : {fact_sales.count()} rows")

# ── Verify ────────────────────────────────────────────────────
spark.sql("SHOW TABLES").filter(
    "tableName LIKE 'dim_%' OR tableName LIKE 'fact_%'"
).show()

# ── Gold DQ ───────────────────────────────────────────────────
f = spark.table("fact_sales")
assert f.filter(F.col("customer_sk").isNull()).count()    == 0, "null customer_sk"
assert f.filter(F.col("product_sk").isNull()).count()     == 0, "null product_sk"
assert f.filter(F.col("order_date_key").isNull()).count() == 0, "null order_date_key"
print("[DQ] All Gold FK checks passed ✓")
🔍 Lab 4 — SQL Analytics Endpoint Queries 3 validation queries · Run in the SQL endpoint after NB-03 ▾
📊
Query 1 — Sales by product category
Tests: dim_product JOIN · Expected order: Technology ($836K) > Furniture ($742K) > Office Supplies ($719K)
SQL
SQL Category aggregation · SQL endpoint 
SELECT
    p.category,
    ROUND(SUM(f.sales),   0) AS total_sales,
    ROUND(SUM(f.profit),  0) AS total_profit,
    ROUND(AVG(f.profit_margin), 1) AS avg_margin_pct
FROM fact_sales f
JOIN dim_product p ON f.product_sk = p.product_sk
GROUP BY p.category
ORDER BY total_sales DESC;
📅
Query 2 — Revenue by year (growth trend)
Tests: dim_date JOIN · Revenue should grow 2014→2017. If all years null: order_date_key is null → Silver notebook type cast failed
SQL
SQL Year aggregation · SQL endpoint 
SELECT
    d.year,
    COUNT(DISTINCT f.order_id)    AS total_orders,
    ROUND(SUM(f.sales), 0)        AS revenue,
    ROUND(AVG(f.days_to_ship), 1) AS avg_days_to_ship
FROM fact_sales f
JOIN dim_date d ON f.order_date_key = d.date_key
GROUP BY d.year
ORDER BY d.year;
👤
Query 3 — Top 5 customers
Tests: dim_customer JOIN · Sean Miller typically highest-revenue customer in Superstore data
SQL
SQL Top customers · SQL endpoint 
SELECT
    c.customer_name,
    c.segment,
    COUNT(DISTINCT f.order_id)     AS orders,
    ROUND(SUM(f.sales), 0)         AS total_sales,
    ROUND(AVG(f.profit_margin), 1) AS avg_margin_pct
FROM fact_sales f
JOIN dim_customer c ON f.customer_sk = c.customer_sk
GROUP BY c.customer_name, c.segment
ORDER BY total_sales DESC
LIMIT 5;

All 8 DAX measures — complete set
Copy all at once for reference or copy individual measures. Measures 1–4 are simple aggregations. Measures 5–7 use DIVIDE/ROUND. Measure 8 uses CALCULATE + DATEADD time intelligence.
DAX
✓ Total Sales ~$2.30M · Total Profit ~$286K · Total Orders ~5,009 ✓ Profit Margin % ~12.5% · Avg Days to Ship ~3.9 ⚠️ Sales YoY % = BLANK with no year slicer — this is correct 💡 Slice to 2016 → YoY% should show +29.5%
DAX SM_RetailX → fact_sales table → New measure (×8) 
-- ── Measure 1 ────────────────────────────────────────────────
Total Sales = SUM(fact_sales[sales])

-- ── Measure 2 ────────────────────────────────────────────────
Total Profit = SUM(fact_sales[profit])

-- ── Measure 3 ────────────────────────────────────────────────
Total Orders = DISTINCTCOUNT(fact_sales[order_id])

-- ── Measure 4 ────────────────────────────────────────────────
Total Quantity = SUM(fact_sales[quantity])

-- ── Measure 5 ────────────────────────────────────────────────
Profit Margin % =
VAR vSales  = [Total Sales]
VAR vProfit = [Total Profit]
RETURN
    IF(
        ISBLANK(vSales) || vSales = 0,
        BLANK(),
        ROUND(DIVIDE(vProfit, vSales) * 100, 1)
    )

-- ── Measure 6 ────────────────────────────────────────────────
Avg Order Value =
DIVIDE(
    [Total Sales],
    [Total Orders],
    BLANK()
)

-- ── Measure 7 ────────────────────────────────────────────────
Avg Days to Ship = ROUND(AVERAGE(fact_sales[days_to_ship]), 1)

-- ── Measure 8 ────────────────────────────────────────────────
Sales YoY % =
VAR vCurr = [Total Sales]
VAR vPrev =
    CALCULATE(
        [Total Sales],
        DATEADD(dim_date[date], -1, YEAR)
    )
RETURN
    IF(
        ISBLANK(vPrev),
        BLANK(),
        ROUND(DIVIDE(vCurr - vPrev, vPrev) * 100, 1)
    )
